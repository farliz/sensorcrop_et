% Define a script named [serif][nimbus]
\starttypescript [serif] [garamondpro]
  \definefontsynonym [Serif]             [name:GaramondPremrPro]
  \definefontsynonym [SerifBold]         [name:GaramondPremrPro-Bd]
  \definefontsynonym [SerifItalic]       [name:GaramondPremrPro-It]
  \definefontsynonym [SerifSlanted]      [name:GaramondPremrPro-It]
  \definefontsynonym [SerifBoldItalic]   [name:GaramondPremrPro-BdIt]
  \definefontsynonym [SerifBoldSlanted]  [name:GaramondPremrPro-BdIt]
\stoptypescript

% Define a second script named [sans][nimbus]
\starttypescript [sans] [garamondpro]
  \definefontsynonym [Sans]             [name:FiraSans-Regular]
  \definefontsynonym [SansBold]         [name:FiraSans-Bold]
  \definefontsynonym [SansItalic]       [name:FiraSans-Italic]
  \definefontsynonym [SansSlanted]      [name:FiraSans-Italic]
  \definefontsynonym [SansBoldItalic]   [name:FiraSans-BoldItalic]
  \definefontsynonym [SansBoldSlanted]  [name:FiraSans-BoldItalic]
\stoptypescript

% Define the typeface nimbus; use serif for rm-fonts, sans serif for ss-fonts.
\definetypeface [garamondpro]  [rm]  [serif] [garamondpro]
\definetypeface [garamondpro]  [ss]  [sans]  [garamondpro]

% Indicate the typeface to use.
\usetypescript[garamondpro][ec]
